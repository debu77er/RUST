use gtk::prelude::*;
use gtk::{MessageDialog, MessageType, ButtonsType, Window, WindowType};

fn main() {
    // Initialize GTK application
    gtk::init().expect("Failed to initialize GTK.");

    // Create a new top-level window
    let window = Window::new(WindowType::Toplevel);
    window.set_title("Simple Message Box");
    window.set_default_size(300, 100);

    // Connect a signal to handle window close
    window.connect_delete_event(|_, _| {
        gtk::main_quit();
        Inhibit(false)
    });

    // Show the window
    window.show_all();

    // Create a message dialog
    let message_dialog = MessageDialog::new(
        Some(&window),
        gtk::DialogFlags::MODAL,
        MessageType::Info,
        ButtonsType::Ok,
        "Hello! This is a simple message box.",
    );

    // Run the message dialog
    message_dialog.run();

    // Destroy the dialog after it's closed
    message_dialog.destroy();

    // Start the GTK main event loop
    gtk::main();
}
